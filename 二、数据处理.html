import pandas as pd
import requests
import time
def get_geocode(address, ak):
    url = 'http://api.map.baidu.com/geocoding/v3/'
    params = {
        'address': address,
        'output': 'json',
        'ak': ak
    }
    response = requests.get(url, params=params)
    data = response.json()
    if data['status'] == 0:
        location = data['result']['location']
        return location['lng'], location['lat']
    else:
        return None
    time.sleep(0.5)

ak = '2XXwGmMnh2yoi8GHhQZrhVzLxENnetyB'

# 添加经纬度列
data['经度'] = None
data['纬度'] = None

# 遍历数据集中的每行院校名称
for index, row in data.iterrows():
    college = row['院校名称']
    try:
        lng, lat = get_geocode(college, ak)
        if lng and lat:
            data.at[index, '经度'] = lng
            data.at[index, '纬度'] = lat
            print(college, ':', lng, lat)
            
    except:
        data.at[index, '经度'] = '未知'
        data.at[index, '纬度'] = '未知'
        print(college, ':', "未知", "未知")
